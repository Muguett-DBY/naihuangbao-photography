### 3.2 提取通用 i18n 工具
- 创建 `src/lib/i18n-helpers.ts`:
  ```ts
  export function getLocalizedField(obj: any, lang: string, field: string) {
    const key = `${field}_${lang}`;
    if (lang !== "zh-CN" && obj[key]) return obj[key];
    return obj[field];
  }
  ```

### 3.3 修复 `formMsg.includes("成功")` 逻辑
